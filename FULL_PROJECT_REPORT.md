async def create_user(telegram_id: int, referrer_id: int = None):
    user = User(telegram_id=telegram_id)
    
    if referrer_id:
        referrer = await get_user(referrer_id)
        if referrer:
            user.referrer_id = referrer.id
            referrer.referral_count += 1
            
            # Бонус: 3 дня VIP
            referrer.vip_until += timedelta(days=3)
    
    return user
